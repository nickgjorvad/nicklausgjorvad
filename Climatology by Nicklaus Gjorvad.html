def read_data(year, flag): #takes two parameters year and flag
    if flag == True:
        file = open("air_temp." + str(year), "r")
    else:
        file = open("precip." + str(year), "r")
        
    lst = []
    for i in file:
        y = i.split()
        for a in range(len(y)):
            y[a] = float(y[a]) #turns element to a float
        y.insert(0, int(year)) # adds the year first to the list
        y.insert(3, min(y[3:])) # starting at the fourth element adds to the list
        y.insert(4, max(y[4:])) # starting at the fifth element adds to the list
        y.insert(5, round(sum(y[5:])/len(y[5:]), 2)) #round to the second decimal average
        y.insert(6, round(sum(y[6:]), 2)) #round to the second decimal total
        
        lst.append(y)
    return(lst)

def get_located_between(lst, col, min_value, max_value): #takes four parameters. 
    num = [] #blank list created
    for line in lst: #for loop that goes through the list 
        if line[col] >= min_value and line[col] < max_value: #finds the values between the minimum and maximum.
            num.append(line) #adds the element to the end.
    return(num) #returns the number

def tropics(lst, year): # takes two parameters
    x1 = get_month_names() #uses the get_month_names function from above 
    top = get_located_between(lst, WeatherData.lat, 23.43684, 90) #finds the data between 23 and 90 aka Cancer
    y1 = [] #creates a blank list 
    for i in range(0,12,1): #goes through all of the months in increments of 1
        mean = get_column_values(top, i)
        mean1 = statistics.mean(mean)#finds the mean of every month
        y1.append(mean1)
    
    x2 = get_month_names()
    bottom = get_located_between(lst, WeatherData.lat, -90, 23.43684) #finds the data between -90 and 23, aka Capricorn
    y2 = []
    for i in range(0,12,1): #goes through all of the months in increments of 1
        mean = get_column_values(bottom, i)
        mean2 = statistics.mean(mean)#finds the mean of every month
        y2.append(mean2)
    
    x3 = get_month_names()
    middle = get_located_between(lst, WeatherData.lat, -23.43684, 23.43684)#finds the data between -23 and 23, everything between Capricorn and Cancer
    y3 = []
    for i in range(0,12,1): #goes through all of the months in increments of 1
        mean = get_column_values(middle, i)
        mean3 = statistics.mean(mean) #finds the mean of every month
        y3.append(mean3)
    
    
                
    plt.plot(x1, y1, label = "Above the Tropic of Cancer")

def percent(data, flag, year): #takes three parameters
    plt.figure(figsize = (10, 5))
    months = get_month_names() #gets the x axis values
    x1 = months
    
    y1 = []
    y2 = []
    y3 = []
    
    for w in range(WeatherData.jan, WeatherData.dec + 1): #for loop that 
        print(w)
        x = get_column_values(data, w) #gets the temperature from the month and sorts it from biggest to smallest
        x.sort()
        y1.append(max(x)) #finds the max temp and appends to it
        inde = int(round(len(x)) * .99) #finds the 99th percentile and rounds it
        y2.append(x[inde]) 
        inde1 = int(round(len(x)) * .50) #finds the 50th percentile and rounds it
        y3.append(x[inde1])
        
    plt.bar(x1, y1, label = "Maximum")
    plt.bar(x1, y2, label = "99th Percentile")
    plt.bar(x1, y3, label = "50th Percentile")
    plt.legend(loc = "center left", bbox_to_anchor = (1, 0.5))
    plt.xlabel("Months")
    
    if flag:
        plt.title("Temperture Observations Per Month in Year " + str(year))
        plt.ylabel("Temperature")
        plt.show()
    else:
        plt.title("Precipitation Observations Per Month in Year" + str(year))
        plt.ylabel("Precipitation")
        plt.show()

def heat(data, flag, year):
    plt.figure(figsize = (10,5))
    longitude = []
    latitude = []
    column = []
    
    plt.xlabel("Longitude")
    plt.ylabel("Latitude")
    
    for tp in data:
        longitude.append(tp[WeatherData.long]) 
        latitude.append(tp[WeatherData.lat]) 
        column.append(tp[WeatherData.annual_avg]) 
    
    if flag:
        plt.title("Heat Map of Temperature Values in Year " + str(year))
        plt.scatter(longitude, latitude, c = column)
        plt.colorbar()
        plt.show()
    else:
        plt.title("Heat Map of Precipitation Values in Year " + str(year))
        plt.scatter(longitude, latitude, c = column, norm = colors.LogNorm())
        plt.colorbar()
        plt.show()

def pie_graph(lst, year):
    months = ["first", "second", "third", "fourth"]
    explode = (0.1, 0, 0, 0)
    
    
    first = get_located_between(lst, WeatherData.lat, 0, 90) #finds the data between the numbers
    second = get_located_between(lst, WeatherData.lat, -90, 0) 
    third = get_located_between(lst, WeatherData.lat, 90, 180) 
    fourth = get_located_between(lst, WeatherData.lat, -90, 180) 
    
    total = [len(first), len(second), len(third), len(fourth)] #plots the data
    
    plt.pie(total, explode = explode, labels = months, autopct = "%1.1f%%", shadow = True, startangle = 90)
